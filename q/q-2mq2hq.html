图片去重

// 基于图片 URL 的 MD5 去重
func DeduplicateImages(images []string) []string {
    seen := make(map[string]bool)
    result := make([]string, 0)
    for _, url := range images {
        hash := md5.Sum([]byte(url))
        key := hex.EncodeToString(hash[:])
        if !seen[key] {
            seen[key] = true
            result = append(result, url)
        }
    }
    return result
}
