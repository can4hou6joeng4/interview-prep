用 Go 实现生产者-消费者模型（channel + select）

简单 

func main() {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    queue := make(chan int, 16)
    var wg sync.WaitGroup

    // producer
    wg.Add(1)
    go func() {
        defer wg.Done()
        for i := 0; i < 100; i++ {
            select {
            case queue <- i:
            case <-ctx.Done():
                return
            }
        }
        close(queue)
    }()

    // consumers
    for i := 0; i < 3; i++ {
        wg.Add(1)
        go func(worker int) {
            defer wg.Done()
            for {
                select {
                case v, ok := <-queue:
                    if !ok {
                        return
                    }
                    fmt.Println(worker, v)
                case <-ctx.Done():
                    return
                }
            }
        }(i)
    }

    wg.Wait()
}
这题别只写 happy path，最好带上 close(channel)、context 和优雅退出，面试官会更认可。
