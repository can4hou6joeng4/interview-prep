多租户隔离方案对比

  • 独立数据库：隔离性最强，成本最高，适合大客户
  • 共享数据库 + 独立 Schema：中等隔离，运维复杂度中等
  • 共享数据库 + 共享表 + StoreId 区分：成本最低，需要严格的数据隔离逻辑 ← 你的方案

// 动态切换数据库连接
func GetDBByStoreId(storeId uint) *gorm.DB {
    connKey := fmt.Sprintf("store_%d", storeId)
    if db, ok := dbPool.Load(connKey); ok {
        return db.(*gorm.DB)
    }
    // 根据 storeId 路由到对应数据源
    dsn := getStoreDSN(storeId)
    db := initDB(dsn)
    dbPool.Store(connKey, db)
    return db
}
